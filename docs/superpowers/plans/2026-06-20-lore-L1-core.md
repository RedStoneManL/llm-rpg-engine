# Lore L1 (Core LoreSystem) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`) syntax.

**Goal:** The core event-line mechanism: a `LoreSystem` holding lore lines (skeleton + stages), a per-turn seeded 暗骰 that independently advances each active line a stage and drops a clue, and an inject that surfaces those clues subtly for the narrator.

**Architecture:** `LoreSystem` (ContextSystem) owns harness-authored `lore_created`/`lore_advanced` events + a `{lines:{}}` slice. `loop/lore.py` provides `create_lore_line(...)` (append a skeleton) and `run_lore(...)` (a post-apply fleet hook mirroring `run_director`: per active line, a seeded `Oracle.d100()` vs the line's threshold; pass → advance a stage + emit `lore_advanced` with that stage's clue). `run_turn` calls `run_lore` after the other fleet hooks, guarded by `registry.owner_of_event("lore_advanced")`.

**Tech Stack:** Python 3.12, pytest (offline/deterministic). Run: `cd /root/rpg-engine-app && PYTHONPATH=/root/rpg-engine-app python3 -m pytest -q`.

**Spec:** `docs/superpowers/specs/2026-06-20-lore-event-line-design.md` (§"L1 详细设计"). This is phase L1 of 4; lifespans/dormancy (L2), density-generation (L3), complex end-states (L4) are OUT of scope.

```python
"""systems.lore — LoreSystem (L1): event-line entities + per-stage clues.

Harness-authored only (no commit section). loop/lore.create_lore_line appends
lore_created (a pre-generated skeleton); loop/lore.run_lore appends lore_advanced
each turn a line's seeded 暗骰 passes its threshold. inject surfaces active lines'
latest clues for the narrator to weave in subtly (never as an explicit quest).

Slice world["systems"]["lore"] = {"lines": {<id>: {
    complexity, about, secret, anchor, stages:[{hint,...}], threshold,
    stage_idx (int, -1 = not started), status ("active"|"resolved"|"expired"),
    clues_dropped:[str]}}}

L1 scope: lifecycle/dormancy (L2), density-generation (L3), complex end-states (L4)
are separate phases. Rewind-safe: the slice folds from events.
"""
from __future__ import annotations

from kernel.contextsystem import ContextSystem, Fragment
from engine.log import get_logger

log = get_logger("systems.lore")


class LoreSystem(ContextSystem):
    name = "lore"

    def requires(self) -> set[str]:
        return {"ontology"}

    def event_types(self) -> set[str]:
        return {"lore_created", "lore_advanced"}

    def commit_sections(self) -> set[str]:
        return set()

    def empty_state(self) -> dict:
        return {"lines": {}}

    def apply(self, world: dict, event: dict) -> None:
        t = event["type"]
        d = event.get("deltas", {})
        lines = world["systems"][self.name]["lines"]

        if t == "lore_created":
            lid = d.get("id")
            if not lid:
                log.warning("lore_created missing id; skipped (%s)", event.get("id"))
                return
            lines[lid] = {
                "complexity": d.get("complexity"),
                "about": d.get("about"),
                "secret": d.get("secret"),
                "anchor": d.get("anchor"),
                "stages": d.get("stages", []),
                "threshold": d.get("threshold", 50),
                "stage_idx": -1,
                "status": "active",
                "clues_dropped": [],
            }
            log.debug("lore_created id=%s complexity=%s", lid, lines[lid]["complexity"])
            return

        if t == "lore_advanced":
            lid = d.get("id")
            ln = lines.get(lid)
            if ln is None:
                log.warning("lore_advanced for unknown line %s; skipped", lid)
                return
            si = d.get("stage_idx")
            if si is not None:
                ln["stage_idx"] = si
            hint = d.get("hint")
            if hint:
                ln["clues_dropped"].append(hint)
            log.debug("lore_advanced id=%s → stage=%s", lid, ln["stage_idx"])
            return

    def inject(self, scene: dict, world: dict) -> Fragment | None:
        lines = (world.get("systems", {}).get(self.name) or {}).get("lines", {})
        clues = [ln["clues_dropped"][-1] for ln in lines.values()
                 if ln.get("status") == "active" and ln.get("clues_dropped")]
        if not clues:
            return None
        text = ("〔环境暗线·可微妙织入,勿点破为任务〕\n"
                + "\n".join(f"  · {c}" for c in clues))
        return Fragment(system="lore", layer="scene", text=text,
                        affordance="可将上述暗线线索作为环境细节自然带出,但不要明示为任务")
```

```python
"""loop.lore — lore creation helper + the per-turn 暗骰 hook (L1).

create_lore_line: validate a pre-generated skeleton, append a lore_created event.
run_lore: post-apply fleet hook (mirrors loop.director.run_director). For each
active line, a seeded Oracle.d100() (rewind-safe via scene_seed) vs the line's
threshold; on a pass, advance one stage and emit lore_advanced carrying that
stage's clue. Cheap, no LLM, deterministic. Guarded by registry ownership so it
is a clean no-op when LoreSystem is not registered.
"""
from __future__ import annotations

from engine.oracle import Oracle, scene_seed
from kernel.events import kernel_event
from kernel.projection import project  # noqa: F401  (caller re-projects; imported for parity)
from engine.log import get_logger

log = get_logger("loop.lore")

_REQUIRED = ("id", "complexity", "about", "stages", "threshold")


def create_lore_line(store, skeleton: dict, *, day: int, scene: str, turn: int) -> dict:
    """Validate a skeleton and append a lore_created event. Returns the event."""
    missing = [k for k in _REQUIRED if k not in skeleton]
    if missing:
        raise ValueError(f"lore skeleton missing required fields: {missing}")
    ev = kernel_event("lore_created", day=day, scene=scene,
                      summary=f"暗线生成:{skeleton['id']}",
                      deltas=dict(skeleton), turn=turn)
    store.append(ev)
    log.debug("create_lore_line id=%s complexity=%s", skeleton["id"], skeleton.get("complexity"))
    return ev


def run_lore(registry, store, world: dict) -> list[dict]:
    """Per-turn seeded 暗骰: advance each active line whose roll passes threshold."""
    if registry.owner_of_event("lore_advanced") is None:
        return []  # LoreSystem not registered → clean no-op
    lines = (world.get("systems", {}).get("lore") or {}).get("lines", {})
    if not lines:
        return []

    events = list(store.iter_events())
    next_turn = max((e.get("turn") or 0 for e in events), default=0) + 1
    campaign_seed = (world.get("meta", {}) or {}).get("campaign_seed", 0)
    day = events[-1]["day"] if events else 1
    scene = (world.get("meta", {}) or {}).get("scene") or "scene"

    appended: list[dict] = []
    for lid, ln in lines.items():
        if ln.get("status") != "active":
            continue
        stages = ln.get("stages", [])
        idx = ln.get("stage_idx", -1)
        if idx + 1 >= len(stages):
            continue  # at last stage; resolution is L4, not L1
        seed = scene_seed(campaign_seed, lid, next_turn)
        roll = Oracle(seed).d100()  # 1..100
        if roll <= ln.get("threshold", 50):
            new_idx = idx + 1
            st = stages[new_idx]
            hint = st.get("hint") if isinstance(st, dict) else str(st)
            ev = kernel_event("lore_advanced", day=day, scene=scene,
                              summary=f"暗线推进:{lid}→stage{new_idx}",
                              deltas={"id": lid, "stage_idx": new_idx, "hint": hint},
                              turn=next_turn)
            store.append(ev)
            appended.append(ev)
            log.debug("run_lore: %s advanced to stage %d (roll=%d<=%d)",
                      lid, new_idx, roll, ln.get("threshold", 50))
    return appended
```
